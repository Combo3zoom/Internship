import Validation
from Iterator import IterRandom


def RandomValue(function):
    def RandomValueForSpesialFunction(listt):
        lowerLimit = Validation.isIntNumber(input('Enter integer value lower limit: '))
        upperLimit = Validation.isIntNumber(input('Enter integer value upper limit: '))
        lowerLimit, upperLimit = Validation.rangeLimits(lowerLimit, upperLimit)
        return function(listt, lowerLimit, upperLimit)

    return RandomValueForSpesialFunction

class RandomNumber:
    @staticmethod
    @RandomValue
    def JustRandomValue(listt, lowerLimit = None, upperLimit = None):
        listt.randomAdd(lowerLimit, upperLimit)
        return listt

    @staticmethod
    @RandomValue
    def IterRandomValue(listt, lowerLimit = None, upperLimit = None):
        val = IterRandom(lowerLimit, upperLimit)
        listt.add(val)
        return listt
