def isFloatNumber(value = None):
    try:
        value = float(value)
        return value
    except ValueError:
        print("Неправильний тип")
        raise ValueError

def isIntNumber(value = None):
    try:
        value = int(value)
        return value
    except ValueError:
        print("Неправильний тип")
        raise ValueError

def CheckSizeOrIndex(value):
    try:
        value = isIntNumber(value)
        if value >= 0:
            return value
    except ValueError:
        raise ValueError



def inputMenu(number):
    try:
        lower,upper = 1,6
        if number == 'exit':
            return number
        number = isIntNumber(number)
        if(number >= lower or number <= upper):
            return number
    except ValueError:
        print("Uncorrect type")
        raise ValueError
    except Exception:
        print("Enter numeric range from 1 to 6")
        raise Exception
    raise Exception

def rangeLimits(lower, upper):
    if(upper < lower):
        upper, lower = lower, upper
    return lower, upper
