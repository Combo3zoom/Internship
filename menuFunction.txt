def Option1(listt):
    file = input("Enter path file: ")
    strategy = FileStrategy(file)
    value = strategy.execute()
    index = Validation.CheckSizeOrIndex((input('Enter integer value, which greater zero: ')))
    for i in range(len(value)):
        listt.insert(value[i], index)
        index += 1
    return listt


def Option2(listt):
    while True:
        lowerLimit = Validation.isIntNumber(input('Enter integer value lower limit: '))
        upperLimit = Validation.isIntNumber(input('Enter integer value upper limit: '))
        if lowerLimit == "exit" or upperLimit == "exit":
            break
        strategy = IteratorStrategy(lowerLimit, upperLimit)
        listt.insert(strategy.execute(),
                     Validation.CheckSizeOrIndex((input('Enter integer value, which greater zero: '))))
    return listt
