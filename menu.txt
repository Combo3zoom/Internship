import Validation
from main import LinkedList

def menu():
    listt = LinkedList()
    while True:
        try:
            print('Enter \n'
                  '1 - add value \n'
                  '2 - autogeneration list \n'
                  '3 - insert value by index \n'
                  '4 - delete by index \n'
                  '5 - show list \n'
                  '6 - delete recurrences for a specific item \n'
                  'exit - exit the program \n')
            userChoose = Validation.inputMenu(input('Enter from 1 to 6 or exit: '))
            if userChoose == 1:
                listt.add(Validation.isIntNumber(input('Enter integer value: ')))
                continue
            if userChoose == 2:
                lowerLimit = Validation.isIntNumber(input('Enter integer value lower limit: '))
                upperLimit = Validation.isIntNumber(input('Enter integer value upper limit: '))
                #count = Validation.CheckSizeOrIndex(input("Enter count elements in list"))
                lowerLimit, upperLimit = Validation.rangeLimits(lowerLimit, upperLimit)
                listt.randomAdd(lowerLimit, upperLimit)
                continue
            if userChoose == 3:
                listt.insert(Validation.isIntNumber(input('Enter integer value: ')), Validation.CheckSizeOrIndex((input('Enter integer value, which greater zero: '))))
                continue
            if userChoose == 4:
                listt.deleteNode(Validation.CheckSizeOrIndex(input('Enter index value, which you want delete ')))
                continue
            if userChoose == 5:
                listt.printList()
                continue
            if userChoose == 6:
                listt.DeleteReplaceNumbers(Validation.isIntNumber(input('Enter value, which you want delete in list ')))
                continue
            if userChoose == 'exit':
                break
            if userChoose != 'exit':
                continue
        except Exception:
            continue




    return

menu()
